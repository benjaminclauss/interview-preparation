from collections import deque

numbers = deque()

# Use append like before to add elements.
numbers.append(99)
numbers.append(15)

# You can pop like a stack.
last_item = numbers.pop()

class Stack:
    def __init__(self):
        self.stack = []

    def pop(self):
        if len(self.stack) < 1:
            return None
        return self.stack.pop()

    def push(self, item):
        self.stack.append(item)

from collections import deque

numbers = deque()

# Use append like before to add elements.
numbers.append(99)
numbers.append(15)

# You can pop like a stack.
last_item = numbers.popleft()

class Queue:
    def __init__(self):
        self.queue = []

    def enqueue(self, item):
        self.queue.append(item)

    def dequeue(self):
        if len(self.queue) < 1:
            return None
        return self.queue.pop(0)

    def size(self):
        return len(self.queue)
